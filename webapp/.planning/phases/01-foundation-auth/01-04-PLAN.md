    Create `src/actions/projects.ts` with createProject and deleteProject.

    The 6 GBD workflows and their initial states per CONTEXT.md:
    - `start`: available (entry point)
    - `platform`: locked (depends on start)
    - `campaign`: locked (depends on platform)
    - `site-standalone`: available (independent chain)
    - `wireframe`: locked (depends on site-standalone)
    - `site`: locked (depends on site-standalone — the full site workflow)

    Note: CONTEXT.md mentions 6 workflows: /start, /platform, /campaign, /site, /site-standalone, /wireframe.

    ```typescript
    "use server"

    import { createClient } from "@/lib/supabase/server"
    import { revalidatePath } from "next/cache"
    import { redirect } from "next/navigation"

    const INITIAL_WORKFLOW_STATES = [
      { slug: "start",           status: "available" },
      { slug: "platform",        status: "locked" },
      { slug: "campaign",        status: "locked" },
      { slug: "site-standalone", status: "available" },  // independent chain
      { slug: "site",            status: "locked" },
      { slug: "wireframe",       status: "locked" },
    ]

    export async function createProject(name: string) {
      const supabase = await createClient()

      const { data: project, error } = await supabase
        .from("projects")
        .insert({ name: name.trim() })
        .select()
        .single()

      if (error || !project) throw new Error(error?.message ?? "Failed to create project")

      // Initialize workflow states for the new project
      const workflowRows = INITIAL_WORKFLOW_STATES.map((w) => ({
        project_id: project.id,
        workflow_slug: w.slug,
        status: w.status,
      }))

      await supabase.from("workflow_states").insert(workflowRows)

      revalidatePath("/dashboard")
      redirect(`/projects/${project.id}`)
    }

    export async function deleteProject(id: string) {
      const supabase = await createClient()

      await supabase.from("projects").delete().eq("id", id)
      // workflow_states cascade-delete via FK ON DELETE CASCADE

      revalidatePath("/dashboard")
      redirect("/dashboard")
    }
    ```

    Create `src/components/create-project-modal.tsx` — Client Component modal for project creation:
    ```typescript
    "use client"

    import { useState, useTransition } from "react"
    import { createProject } from "@/actions/projects"

    export function CreateProjectModal() {
      const [open, setOpen] = useState(false)
      const [name, setName] = useState("")
      const [error, setError] = useState<string | null>(null)
      const [isPending, startTransition] = useTransition()

      function handleSubmit(e: React.FormEvent) {
        e.preventDefault()
        const trimmed = name.trim()
        if (!trimmed) {
          setError("Le nom du projet est requis.")
          return
        }
        setError(null)
        startTransition(async () => {
          await createProject(trimmed)
          // redirect() in createProject navigates away automatically
        })
      }

      return (
        <>
          <button
            onClick={() => { setOpen(true); setName(""); setError(null) }}
            className="px-4 py-2 bg-gray-900 text-white text-sm font-medium rounded hover:bg-gray-700 transition-colors"
          >
            Nouveau projet
          </button>

          {open && (
            <div className="fixed inset-0 bg-black/40 flex items-center justify-center z-50">
              <div className="bg-white rounded-lg shadow-lg p-6 w-full max-w-sm">
                <h2 className="text-lg font-semibold mb-4">Nouveau projet</h2>
                <form onSubmit={handleSubmit} className="space-y-4">
                  <div>
                    <label htmlFor="project-name" className="block text-sm font-medium text-gray-700 mb-1">
                      Nom du projet
                    </label>
                    <input
                      id="project-name"
                      type="text"
                      value={name}
                      onChange={(e) => setName(e.target.value)}
                      placeholder="Ex: Acme Corp"
    ```

    Create `src/app/dashboard/page.tsx` — Server Component fetching projects with workflow counts. Per CONTEXT.md: simple list layout (not cards/table), each row shows name + date + "X/6 workflows", ordered by updated_at DESC:
    ```typescript
    import { createClient } from "@/lib/supabase/server"
    import { signOut } from "@/actions/auth"
    import { CreateProjectModal } from "@/components/create-project-modal"
    import Link from "next/link"
    import { deleteProject } from "@/actions/projects"

    function formatDate(dateString: string): string {
      return new Date(dateString).toLocaleDateString("fr-FR", {
        day: "numeric",
        month: "short",
        year: "numeric",
      })
    }

    export default async function DashboardPage() {
      const supabase = await createClient()

      const { data: projects } = await supabase
        .from("projects")
        .select(`
          id,
          name,
          updated_at,
          workflow_states(status)
        `)
        .order("updated_at", { ascending: false })

      const projectsWithCount = (projects ?? []).map((p) => ({
        ...p,
        completedCount: p.workflow_states.filter((w: { status: string }) => w.status === "completed").length,
      }))

      return (
        <div className="min-h-screen bg-gray-50">
          <header className="bg-white border-b border-gray-200 px-6 py-4 flex items-center justify-between">
            <h1 className="text-lg font-semibold">GET BRANDON</h1>
            <form action={signOut}>
              <button type="submit" className="text-sm text-gray-500 hover:text-gray-900 transition-colors">
                Déconnexion
              </button>
            </form>
          </header>

          <main className="max-w-3xl mx-auto px-6 py-8">
            <div className="flex items-center justify-between mb-6">
              <h2 className="text-xl font-semibold">Projets</h2>
              <CreateProjectModal />
            </div>

            {projectsWithCount.length === 0 ? (
              <div className="text-center py-16 text-gray-500">
                <p className="mb-4">Aucun projet.</p>
                <CreateProjectModal />
              </div>
            ) : (
              <ul className="divide-y divide-gray-200 bg-white rounded-lg shadow-sm">
                {projectsWithCount.map((project) => (
                  <li key={project.id} className="flex items-center justify-between px-4 py-3 hover:bg-gray-50">
                    <Link
                      href={`/projects/${project.id}`}
                      className="flex-1 min-w-0"
                    >
                      <span className="font-medium text-gray-900 truncate block">{project.name}</span>
                      <span className="text-sm text-gray-500">
                        {formatDate(project.updated_at)} · {project.completedCount}/6 workflows
                      </span>
                    </Link>
                    <form
                      action={async () => {
                        "use server"
                        await deleteProject(project.id)
                      }}
                      className="ml-4"
                    >
                      <button
                        type="submit"
                        onClick={(e) => {
                          if (!confirm("Supprimer ce projet ? Cette action est irréversible.")) {
                            e.preventDefault()
                          }
                        }}
                        className="text-sm text-red-500 hover:text-red-700 transition-colors"
                      >
                        Supprimer
                      </button>
                    </form>
                  </li>
                ))}
              </ul>
            )}
          </main>
        </div>
      )
    }
    ```

    Note on deleteProject inline form action: Next.js 16 supports inline `"use server"` in Server Components. However if TypeScript complains about the inline action, move the delete call to a separate named function or use a Client Component wrapper. The import of `deleteProject` at the top can be used directly in the inline action.

