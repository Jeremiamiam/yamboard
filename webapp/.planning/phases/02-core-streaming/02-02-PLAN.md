```typescript
import Anthropic from "@anthropic-ai/sdk"
import { createClient } from "@/lib/supabase/server"
import { persistMessage } from "@/lib/supabase/messages"

// REQUIRED: Edge runtime is incompatible with Anthropic SDK — use Node.js
export const runtime = "nodejs"
// REQUIRED: Prevent Next.js from caching the streaming response
export const dynamic = "force-dynamic"
// REQUIRED: 5 minutes — matches GBD workflow duration (CHAT-03)
export const maxDuration = 300

// Instantiated once (module-level) — reads ANTHROPIC_API_KEY from env automatically
const anthropic = new Anthropic()

// GBD generic system prompt for Phase 2 (workflow-specific prompts in Phase 3)
const GBD_SYSTEM_PROMPT = `Tu es Brandon, un assistant IA spécialisé dans la création de stratégies de marque et de contenu marketing. Tu aides les équipes de l'agence Yam à conduire des projets clients structurés via des workflows dédiés (/start, /platform, /campaign, /site-standalone, /wireframe).

Réponds toujours en français sauf si l'utilisateur écrit dans une autre langue. Sois précis, professionnel et orienté résultats.`

export async function POST(request: Request) {
  // Auth validation — uses getClaims() internally (no network call, Phase 1 pattern)
  const supabase = await createClient()
  const {
    data: { user },
  } = await supabase.auth.getUser()
  if (!user) {
    return new Response("Unauthorized", { status: 401 })
  }

  let projectId: string
  let message: string
  let history: Array<{ role: "user" | "assistant"; content: string }>

  try {
    const body = await request.json()
    projectId = body.projectId
    message = body.message
    history = body.history ?? []
  } catch {
    return new Response("Invalid JSON body", { status: 400 })
  }

  if (!projectId || !message) {
    return new Response("Missing projectId or message", { status: 400 })
  }

  // Build messages array: full history + new user message
  // Anthropic API is stateless — must pass full context each time (RESEARCH pitfall 3)
  const apiMessages: Array<{ role: "user" | "assistant"; content: string }> = [
    ...history,
    { role: "user", content: message },
  ]

  // Create MessageStream — .toReadableStream() produces spec-compliant ReadableStream
  // Accumulate full response to persist after stream completes
  const stream = anthropic.messages.stream({
    model: "claude-sonnet-4-6",
    max_tokens: 8192,
    system: GBD_SYSTEM_PROMPT,
    messages: apiMessages,
  })

  // Persist assistant message after stream completes (server-side, after message_stop)
  // Fire-and-forget with error logging — do NOT await in the stream path
  stream.on("message", async (finalMessage) => {
    const assistantContent = finalMessage.content
      .filter((block) => block.type === "text")
      .map((block) => (block as { type: "text"; text: string }).text)
      .join("")

    if (assistantContent) {
      try {
        await persistMessage(projectId, "assistant", assistantContent, supabase)
      } catch (err) {
        console.error("[chat/route] failed to persist assistant message:", err)
      }
    }
  })

  // Return the raw SSE stream as the response body
  // Headers required to prevent proxy/CDN buffering (RESEARCH pitfall 2)
  return new Response(stream.toReadableStream(), {
    headers: {
      "Content-Type": "text/event-stream; charset=utf-8",
      "Cache-Control": "no-cache, no-transform",
      "Connection": "keep-alive",
      "X-Accel-Buffering": "no", // nginx anti-buffering (defensive)
    },
  })
}
```

Key implementation notes:
- `anthropic.messages.stream()` NOT `anthropic.messages.create()` — stream() provides the MessageStream with `.toReadableStream()`
- `stream.on("message", ...)` fires after `message_stop` — this is when the complete assistant message is available (RESEARCH pitfall 4: do NOT persist mid-stream)
- `runtime = "nodejs"` is mandatory — do NOT use `"edge"` (Anthropic SDK incompatible with Edge runtime)
- `dynamic = "force-dynamic"` prevents Next.js caching of the SSE response
- `maxDuration = 300` covers 5-minute GBD workflows (CHAT-03)
- `ANTHROPIC_API_KEY` must be in `.env.local` — the Anthropic() constructor reads it automatically. If not present, add `ANTHROPIC_API_KEY=sk-ant-...` to `.env.local`
