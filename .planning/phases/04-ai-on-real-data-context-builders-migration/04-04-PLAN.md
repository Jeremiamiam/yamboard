    ```typescript
    const TOKEN_BUDGETS = { agency: 20_000, client: 30_000, project: 40_000 } as const

    function estimateTokens(text: string): number {
      return Math.ceil(text.length / 3.5) // Conservative for French text
    }

    function enforceTokenBudget(
      context: string,
      contextType: 'agency' | 'client' | 'project',
      // Rebuild functions for each truncation level
      rebuildLevel1: () => string,
      rebuildLevel2: () => string,
      rebuildLevel3: () => string,
    ): string {
      const budget = TOKEN_BUDGETS[contextType]
      if (estimateTokens(context) <= budget) return context

      console.warn(`[context-builders] ${contextType} over budget (${estimateTokens(context)} est. tokens), applying level 1 truncation`)
      const l1 = rebuildLevel1()
      if (estimateTokens(l1) <= budget) return l1

      console.warn(`[context-builders] ${contextType} still over budget after level 1, applying level 2 (doc truncation)`)
      const l2 = rebuildLevel2()
      if (estimateTokens(l2) <= budget) return l2

      console.warn(`[context-builders] ${contextType} still over budget after level 2, applying level 3 (description truncation)`)
      return rebuildLevel3()
    }
    ```

    Note: Alternatively, implement truncation inline in each builder using a simpler pattern — pass a `truncLevel: 0 | 1 | 2 | 3` parameter and rebuild if needed. The exact algorithm is Claude's Discretion. Keep it deterministic (always truncate docs oldest-to-newest order is already guaranteed by `created_at ASC`).

      // Build sections per client
