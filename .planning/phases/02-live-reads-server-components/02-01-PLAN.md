    Create directory `dashboard/src/lib/data/` and three files. Each wraps mock helpers behind async signatures. These exact signatures are what pages will import — the bodies will be replaced with Supabase queries in plans 02-02/03/04.

    **dashboard/src/lib/data/clients.ts:**
    ```typescript
    import 'server-only'
    import { CLIENTS, type Client, type ClientCategory } from '@/lib/mock'

    export async function getClients(category: ClientCategory = 'client'): Promise<Client[]> {
      return CLIENTS.filter((c) => c.category === category)
    }

    export async function getClient(id: string): Promise<Client | null> {
      return CLIENTS.find((c) => c.id === id) ?? null
    }
    ```

    **dashboard/src/lib/data/projects.ts:**
    ```typescript
    import 'server-only'
    import { PROJECTS, type Project } from '@/lib/mock'

    export async function getClientProjects(clientId: string): Promise<Project[]> {
      return PROJECTS.filter((p) => p.clientId === clientId)
    }

    export async function getProject(projectId: string): Promise<Project | null> {
      return PROJECTS.find((p) => p.id === projectId) ?? null
    }
    ```

    **dashboard/src/lib/data/documents.ts:**
    ```typescript
    import 'server-only'
    import { DOCUMENTS, BUDGET_PRODUCTS, type Document, type BudgetProduct } from '@/lib/mock'

    export async function getClientDocs(clientId: string): Promise<Document[]> {
      return DOCUMENTS.filter((d) => d.clientId === clientId && !d.projectId)
    }

    export async function getProjectDocs(projectId: string): Promise<Document[]> {
      return DOCUMENTS.filter((d) => d.projectId === projectId)
    }

    export async function getBudgetProducts(projectId: string): Promise<BudgetProduct[]> {
      return BUDGET_PRODUCTS.filter((b) => b.projectId === projectId)
    }
    ```
